-- test program

let x := 10;

let f := \x y ->
    x + y;

let max := \a b ->
    if a > b then a else b;

let min := \a b ->
    if a < b then a else b;

let show := \a s ->
    if a = 0 then
        print 0
    else
        { print s;
          show (a - 1) s
        };


print f 3 (3 + 5);
print (max 1 3);
print min 4 2;

show 3 1
